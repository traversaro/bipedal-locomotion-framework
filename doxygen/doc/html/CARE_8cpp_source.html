122 bool CARE::solve()
123 {
124  const std::size_t n = m_pimpl->A.rows();
125 
126  // compute cholesky decomposition
127  Eigen::LLT<Eigen::MatrixXd> R_cholesky(m_pimpl->R);
128  if (R_cholesky.info() != Eigen::Success)
129  {
130  std::cerr << "[CARE::solve] The matrix R must be positive definite." << std::endl;
131  return false;
132  }
133 
134  // Z represents the Hamiltonian matrix.
135  // _ _
136  // | A B inv(R) B' |
137  // Z = | |
138  // |_Q - A' _|
139 
140  m_pimpl->Z.block(0, 0, n, n) = m_pimpl->A;
141  m_pimpl->Z.block(0, n, n, n) = m_pimpl->B * R_cholesky.solve(m_pimpl->B.transpose());
142  m_pimpl->Z.block(n, 0, n, n) = m_pimpl->Q;
143  m_pimpl->Z.block(n, n, n, n) = -m_pimpl->A.transpose();
144 
145  double relativeNorm = m_pimpl->tolerance;
146  std::size_t iteration = 0;
147  const double p = static_cast<double>(m_pimpl->Z.rows());
148 
149  // run the algorithm
150  for (; iteration < m_pimpl->maxIterations && relativeNorm >= m_pimpl->tolerance; iteration++)
151  {
152  m_pimpl->ZOld = m_pimpl->Z;
153 
154  // R. Byers. Solving the algebraic Riccati equation with the matrix sign
155  // function. Linear Algebra Appl., 85:267–279, 1987
156  // Added determinant scaling to improve convergence (converges in rough half
157  // the iterations with this)
158  const double ck = std::pow(std::abs(m_pimpl->Z.determinant()), -1.0 / p);
159  m_pimpl->Z *= ck;
160  m_pimpl->ZInverse = m_pimpl->Z.inverse();
161  m_pimpl->Z = m_pimpl->Z - 0.5 * (m_pimpl->Z - m_pimpl->ZInverse);
162  relativeNorm = (m_pimpl->Z - m_pimpl->ZOld).norm();
163  }
164 
165  Eigen::Ref<const Eigen::MatrixXd> W11 = m_pimpl->Z.block(0, 0, n, n);
166  Eigen::Ref<const Eigen::MatrixXd> W12 = m_pimpl->Z.block(0, n, n, n);
167  Eigen::Ref<const Eigen::MatrixXd> W21 = m_pimpl->Z.block(n, 0, n, n);
168  Eigen::Ref<const Eigen::MatrixXd> W22 = m_pimpl->Z.block(n, n, n, n);
169 
170  m_pimpl->lhs.topRows(n) = W12;
171  m_pimpl->lhs.bottomRows(n) = W22 + m_pimpl->identity;
172  m_pimpl->rhs.topRows(n) = W11 + m_pimpl->identity;
173  m_pimpl->rhs.bottomRows(n) = W21;
174 
175  Eigen::JacobiSVD<Eigen::MatrixXd> svd(m_pimpl->lhs, Eigen::ComputeThinU | Eigen::ComputeThinV);
176  m_pimpl->solution = svd.solve(m_pimpl->rhs);
177 
178  if (m_pimpl->isVerbose)
179  {
180  std::stringstream info;
181  if (relativeNorm < m_pimpl->tolerance)
182  info << "Solution found.";
183  else
184  info << "Solution not found.";
185  info << std::endl;
186  info << "Number of iteration: " << iteration << ". Relative error: " << relativeNorm << ".";
187 
188  std::cout << "[CARE::solve] " << info.str() << std::endl;
189  }
190 
191  return relativeNorm < m_pimpl->tolerance;
192 }
