16 void BipedalLocomotion::Planners::ContactPhaseList::createPhases()
17 {
18  m_phases.clear();
19 
20  std::map<double, std::unordered_map<std::string, ContactList::const_iterator>> activations, deactivations;
21 
22  for (ContactListMap::iterator list = m_contactLists.begin(); list != m_contactLists.end(); ++list)
23  {
24  const std::string& key = list->first;
25  for (ContactList::const_iterator step = list->second.begin(); step != list->second.end(); ++step)
26  {
27  activations[step->activationTime][key] = step; //By construction, there is only a step given a key and activationTime
28  deactivations[step->deactivationTime][key] = step;
29  }
30  }
31 
32  if (!activations.size())
33  {
34  return;
35  }
36 
37  ContactPhase currentPhase;
38  currentPhase.beginTime = activations.begin()->first;
39  currentPhase.activeContacts = activations.begin()->second;
40 
41  activations.erase(activations.begin());
42 
43  while ((activations.size() + deactivations.size()) > 1)
44  {
45  if ((activations.size() == 0) || (deactivations.begin()->first <= activations.begin()->first))
46  {
47  //Here I need to remove from the current phase the contacts that are going to end
48  currentPhase.endTime = deactivations.begin()->first;
49  m_phases.push_back(currentPhase);
50 
51  currentPhase.beginTime = deactivations.begin()->first;
52 
53  for (auto& toBeRemoved : deactivations.begin()->second)
54  {
55  currentPhase.activeContacts.erase(toBeRemoved.first); //Erase all the contacts which are deactivativated in this instant
56  }
57 
58  deactivations.erase(deactivations.begin());
59 
60  if (activations.size() && (deactivations.begin()->first == activations.begin()->first))
61  {
62  currentPhase.activeContacts.insert(activations.begin()->second.begin(),
63  activations.begin()->second.end()); //Add the new contacts to the list.
64 
65  activations.erase(activations.begin());
66  }
67  }
68  else // (activations.begin()->first < deactivations.begin()->first)
69  {
70  currentPhase.endTime = activations.begin()->first;
71  m_phases.push_back(currentPhase);
72 
73  currentPhase.beginTime = activations.begin()->first;
74  currentPhase.activeContacts.insert(activations.begin()->second.begin(),
75  activations.begin()->second.end()); //Add the new contacts to the list.
76 
77  activations.erase(activations.begin());
78  }
79  }
80 
81  assert(deactivations.size() == 1); //Only one element should be left (deactivations and activations were equal in number, but the head of activations was deleted at the beginning).
82  currentPhase.endTime = deactivations.begin()->first;
83  m_phases.push_back(currentPhase);
84 }
