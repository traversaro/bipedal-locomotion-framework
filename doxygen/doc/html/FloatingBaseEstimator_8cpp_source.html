640 bool FloatingBaseEstimator::updateBaseStateFromIMUState(const FloatingBaseEstimators::InternalState& state,
641  const FloatingBaseEstimators::Measurements& meas,
642  iDynTree::Transform& basePose,
643  iDynTree::Twist& baseTwist)
644 {
645  iDynTree::Position imuPosition;
646  iDynTree::toEigen(imuPosition) = state.imuPosition;
647  iDynTree::Rotation imuRotation;
648  iDynTree::toEigen(imuRotation) = state.imuOrientation.toRotationMatrix();
649  auto A_H_IMU = iDynTree::Transform(imuRotation, imuPosition);
650 
651  iDynTree::Twist v_IMU;
652  iDynTree::Vector3 imuLinVel, imuAngVel;
653  iDynTree::toEigen(imuLinVel) = state.imuLinearVelocity;
654  iDynTree::toEigen(imuAngVel) = state.imuOrientation.toRotationMatrix()*(meas.gyro - state.gyroscopeBias);
655  v_IMU.setLinearVec3(imuLinVel);
656  v_IMU.setAngularVec3(imuAngVel);
657 
658  if (!m_modelComp.getBaseStateFromIMUState(A_H_IMU, v_IMU, basePose, baseTwist))
659  {

667 bool FloatingBaseEstimator::resetEstimator(const Eigen::Quaterniond& newBaseOrientation,
668  const Eigen::Vector3d& newBasePosition)
669 {
670  iDynTree::Rotation Rb;
671  iDynTree::toEigen(Rb) = newBaseOrientation.toRotationMatrix();
672 
673  iDynTree::Position pb;
674  iDynTree::toEigen(pb) = newBasePosition;
675 
676  auto A_H_B = iDynTree::Transform(Rb, pb);
677  iDynTree::Transform IMU_H_RF, IMU_H_LF;
678  iDynTree::JointPosDoubleArray s(m_meas.encoders.size());
679  toEigen(s) = m_meas.encoders;
680 
681  if (!m_modelComp.getIMU_H_feet(s, IMU_H_RF, IMU_H_LF))
682  {
683  std::cerr << "[FloatingBaseEstimator::resetEstimator] Could not reset estimator using new base pose." << std::endl;
684  return false;
685  }
686 
687  auto A_H_IMU = A_H_B*m_modelComp.base_H_IMU();;
688  auto A_H_RF = A_H_IMU*IMU_H_RF;
689  auto A_H_LF = A_H_IMU*IMU_H_LF;
690 
691  // convert to iDynTree Rotation to Eigen angle axis and then to quaternion for consistent conversion
692  Eigen::AngleAxisd imuAngleAxis = Eigen::AngleAxisd(iDynTree::toEigen(A_H_IMU.getRotation()));
693  Eigen::AngleAxisd rfAngleAxis = Eigen::AngleAxisd(iDynTree::toEigen(A_H_RF.getRotation()));
694  Eigen::AngleAxisd lfAngleAxis = Eigen::AngleAxisd(iDynTree::toEigen(A_H_LF.getRotation()));
695 
696  m_state.imuOrientation = Eigen::Quaterniond(imuAngleAxis);
697  m_state.imuPosition = iDynTree::toEigen(A_H_IMU.getPosition());
698  m_state.rContactFrameOrientation = Eigen::Quaterniond(rfAngleAxis);
699  m_state.rContactFramePosition = iDynTree::toEigen(A_H_RF.getPosition());
700  m_state.lContactFrameOrientation = Eigen::Quaterniond(lfAngleAxis);
701  m_state.lContactFramePosition = iDynTree::toEigen(A_H_LF.getPosition());
702 
703  return true;
704 }
