171  return false;
172  }

178  {

198

200  m_kinDyn->getWorldTransform(contactWrench.index()));
201 
202  m_knownCoefficent += m_jacobianMatrix.transpose()
203  * iDynTree::toEigen(contactPtr->getContactWrench());
204  }
205 
206  // add the joint torques to the known coefficent
207  m_knownCoefficent.tail(m_actuatedDoFs) += jointTorques;
208

211
