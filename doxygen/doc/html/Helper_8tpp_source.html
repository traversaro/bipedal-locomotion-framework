179 template <typename T> void mergeSigVector(yarp::sig::Vector& vector, const T& t)
180 {
181  if constexpr (std::is_arithmetic<T>::value)
182  vector.push_back(t);
183

188  "[BipedalLocomotion::YarpUtilities::mergeSigVector] The element "
189  "contained in the vector cannot be converted in a double");

194  vector.push_back(*it);
195 
196  } else if constexpr (has_square_bracket_operator<T>::value)

198  for (int i = 0; i < t.size(); i++)
