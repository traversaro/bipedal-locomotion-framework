588 bool InvariantEKFBaseEstimator::Impl::updateStates(const Eigen::VectorXd& obs,
589  const Eigen::MatrixXd measModelJacobian,
590  const Eigen::MatrixXd& measNoiseVar,
591  const Eigen::MatrixXd& auxMat,
592  FloatingBaseEstimators::InternalState& state,
593  Eigen::MatrixXd& P)
594 {
595  if (measModelJacobian.cols() != P.rows())
596  {
597  std::cerr << "[InvariantEKFBaseEstimator::updateStates] Measurement model Jacobian size mismatch" << std::endl;
598  return false;
599  }
600 
601  if (measModelJacobian.rows() != measNoiseVar.rows())
602  {
603  std::cerr << "[InvariantEKFBaseEstimator::updateStates] Measurement noise covariance matrix size mismatch" << std::endl;
604  return false;
605  }
606 
607  bool estimateBias;
608  if (P.rows() == m_vecSizeWBias)
609  {
610  estimateBias = true;
611  }
612  else
613  {
614  estimateBias = false;
615  }
616 
617  m_PHT = P*measModelJacobian.transpose();
618  m_S = measModelJacobian*m_PHT + measNoiseVar;
619  m_K = m_PHT*(m_S.inverse());
620 
621  // compute innovation delta
622  extractState(state, m_X, m_theta);
623 
624  int copyTimes = obs.rows()/m_X.cols();
625  if (!copyDiagX(m_X, copyTimes, m_BigX))
626  {
627  return false;
628  }
629 
630  m_innovation = m_BigX*obs; // minus bias terms which are zero in reduced form (Section III C of the paper)
631  m_delta = m_K*auxMat*m_innovation;
632 
633  m_deltaTheta.setZero();
634  if (estimateBias)
635  {
636  m_deltaTheta = m_delta.segment<6>(m_vecSizeWOBias);
637  }
638 
639  // update state
640  if (!calcExpHatX(m_delta.segment(0, m_vecSizeWOBias), m_dX))
641  {
642  std::cerr << "[InvariantEKFBaseEstimator::updateStates] Could not compute state update";
643  return false;
644  }
645 
646  if (!constructState( m_dX*m_X, m_theta+m_deltaTheta, state) ) // right invariant update
647  {
648  std::cerr << "[InvariantEKFBaseEstimator::updateStates] Could not update state";
649  return false;
650  }
651  // update covariance
652  m_IminusKH = Eigen::MatrixXd::Identity(P.rows(), P.cols()) - m_K*measModelJacobian;
653  P = m_IminusKH*P*(m_IminusKH.transpose()) + m_K*measNoiseVar*(m_K.transpose());
654  return true;
655 }

751 bool InvariantEKFBaseEstimator::Impl::calcExpHatX(const Eigen::VectorXd& vec,
752  Eigen::MatrixXd& X)
753 {
754  // Exp(vec) = Exp([ w]) = |ExpSO3(w) JlSO3(w)v JlSO3(w)p JlSO3(w)pr JlSO3(w)pl|
755  // ([ a]) | 1 |
756  // ([ v]) | 1 |
757  // ([vr]) | 1 |
758  // ([vl]) | 1|
759  // where JlSO3 is the left Jacobian of SO(3)
760  if (vec.size() != 15)
761  {
762  std::cerr << "[InvariantEKFBaseEstimator::calcExpHatX] State vector does not seem to have expected size of 15x1." << std::endl;
763  return false;
764  }
765 
766  X.resize(7, 7);
767  auto omega = manif::SO3Tangentd(vec.segment<3>(m_vecOffsets.imuOrientation));
768  X.topLeftCorner<3, 3>() = omega.exp().rotation();
769  X.block<3, 1>(0, 3) = omega.ljac()*vec.segment<3>(m_vecOffsets.imuLinearVel);
770  X.block<3, 1>(0, 4) = omega.ljac()*vec.segment<3>(m_vecOffsets.imuPosition);
771  X.block<3, 1>(0, 5) = omega.ljac()*vec.segment<3>(m_vecOffsets.rContactFramePosition);
772  X.block<3, 1>(0, 6) = omega.ljac()*vec.segment<3>(m_vecOffsets.lContactFramePosition);
773 
774  X.bottomLeftCorner<4, 3>().setZero();
775  X.bottomRightCorner<4, 4>() = Eigen::Matrix4d::Identity();
776  return true;
777 }

780  const SkewSymContainter& skew,
