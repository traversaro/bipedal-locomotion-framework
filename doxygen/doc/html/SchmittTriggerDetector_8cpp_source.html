322 void SchmittTriggerUnit::update(const double& currentTime, const double& rawValue)
323 {
324  if (previousTime == initialTime)
325  {
326  (currentTime > initialTime) ? previousTime = initialTime : previousTime = currentTime;
327  }
328 
329  if (!state)
330  {
331  // Check for transition from false to true - if valid over a timeframe, then switch
332  if (rawValue >= params.onThreshold)
333  {
334  (timer >= params.switchOnAfter) ? state = true : timer += (currentTime - previousTime);
335  if (state)
336  {
337  switchTime = currentTime;
338  timer = 0;
339  }
340  }
341  else
342  {
343  timer = 0;
344  }
345  }
346  else
347  {
348  // check for transition from true to false - if valid over a timeframe, then switch
349  if (rawValue <= params.offThreshold)
350  {
351  (timer >= params.switchOffAfter) ? state = false : timer += (currentTime - previousTime);
352  if (!state)
353  {
354  switchTime = currentTime;
355  timer = 0;
356  }
357  }
358  else
359  {
360  timer = 0;
361  }
362  }
363  previousTime = currentTime;
