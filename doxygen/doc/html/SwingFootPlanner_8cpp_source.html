107 bool SwingFootPlanner::updateSE3Traj()
108 {
109  // compute the trajectory at the current time
110  const double shiftedTime = m_currentTrajectoryTime - m_currentContactPtr->deactivationTime;
111 
112  manif::SO3d rotation;
113 
114  // note here we assume that the velocity is expressed using the mixed representation.
115  // for this reason the velocity can be computed splitting the linear and the angular problem.
116  // For the angular part we are interested in the angular velocity expressed in the inertial
117  // frame (right trivialized velocity). For the liner part we are interested on the derivative of
118  // the position vector w.r.t the time. A similar consideration can be done also for the
119  // acceleration.
120 
121  // Note that here we use asSO3() to access to the angular velocity stored in the 6d-vector. (A
122  // similar consideration can be done also for the acceleration.)
123  auto angularVelocity(m_state.mixedVelocity.asSO3());
124  auto angularAcceleration(m_state.mixedAcceleration.asSO3());
125  if (!m_SO3Planner.evaluatePoint(shiftedTime, rotation, angularVelocity, angularAcceleration))
126  {
127  std::cerr << "[SwingFootPlanner::advance] Unable to get the SO(3) trajectory at time t = "
128  << m_currentTrajectoryTime << "." << std::endl;
129  return false;
130  }
131 
132  manif::SE3d::Translation position;
133  if (!m_planarPlanner.evaluatePoint(m_currentTrajectoryTime,
134  position.head<2>(),
135  m_state.mixedVelocity.coeffs().head<2>(),
136  m_state.mixedAcceleration.coeffs().head<2>()))
137  {
138  std::cerr << "[SwingFootPlanner::advance] Unable to get the planar trajectory at time t = "
139  << m_currentTrajectoryTime << "." << std::endl;
140  return false;
141  }
142 
143  if (!m_heightPlanner.evaluatePoint(m_currentTrajectoryTime,
144  position.segment<1>(2),
145  m_state.mixedVelocity.coeffs().segment<1>(2),
146  m_state.mixedAcceleration.coeffs().segment<1>(2)))
147  {
148  std::cerr << "[SwingFootPlanner::advance] Unable to get the height trajectory at time t = "
149  << m_currentTrajectoryTime << "." << std::endl;
150  return false;
151  }
152 
153  // set the transform
154  m_state.transform = manif::SE3d(position, rotation);
155 
156  m_state.isInContact = false;
157 
158  return true;
159 }

161 bool SwingFootPlanner::advance()
162 {
163  // update the time taking into account the limits
164  m_currentTrajectoryTime
165  = std::min(m_dT + m_currentTrajectoryTime, m_contactList.lastContact()->deactivationTime);
166 
167  // here there are several case
168  // 1. the link was already in contact with the environment and now it is still in contact
169  // 2. the link was in contact with the environment but now not (This cannot happen for the
170  // latest contact phase)
171  // 3. the link was not in contact with the environment and now it is still not in contact (This
172  // cannot happen for the latest contact phase)
173  // 4. the link was not in contact with the environment but now it is in contact (This cannot
174  // happen for the latest contact phase)
175 
176  // This is the case (1). In this case we do not have to update the state
177  if (m_state.isInContact && m_currentTrajectoryTime <= m_currentContactPtr->deactivationTime)
178  {
179  return true;
180  }
181 
182  // In theory this should never happen. This check is only for spotting possible bugs
183  if (m_currentContactPtr == m_contactList.end())
184  {
185  std::cerr << "[SwingFootPlanner::advance] During the last phase the link should be in "
186  "contact with the environment."
187  << std::endl;
188  return false;
189  }
190 
191  // This is the case (2)
192  if (m_state.isInContact && m_currentTrajectoryTime > m_currentContactPtr->deactivationTime)
193  {
194  // create a new trajectory in SE(3)
195  const auto nextContactPtr = std::next(m_currentContactPtr, 1);
196  const double T = nextContactPtr->activationTime - m_currentContactPtr->deactivationTime;
197 
198  if (!m_SO3Planner.setRotations(m_currentContactPtr->pose.asSO3(), nextContactPtr->pose.asSO3(), T))
199  {
200  std::cerr << "[SwingFootPlanner::advance] Unable to set the initial and final "
201  "rotations for the SO(3) planner."
202  << std::endl;
203  return false;
204  }
205 
206  if (!m_planarPlanner.setKnots({m_currentContactPtr->pose.translation().head<2>(),
207  nextContactPtr->pose.translation().head<2>()},
208  {m_currentContactPtr->deactivationTime,
209  nextContactPtr->activationTime}))
210  {
211  std::cerr << "[SwingFootPlanner::advance] Unable to set the knots for the planar "
212  "planner."
213  << std::endl;
214  return false;
215  }
216 
217  const double footHeightViaPointPos = (m_currentContactPtr->pose.translation()(2) +
218  nextContactPtr->pose.translation()(2)) / 2.0 + m_stepHeight;
219 
220  const double footHeightViaPointTime = m_footApexTime * T + m_currentContactPtr->deactivationTime;
221 
222  if (!m_heightPlanner.setKnots({m_currentContactPtr->pose.translation().tail<1>(),
223  Vector1d::Constant(footHeightViaPointPos),
224  nextContactPtr->pose.translation().tail<1>()},
225  {m_currentContactPtr->deactivationTime,
226  footHeightViaPointTime,
227  nextContactPtr->activationTime}))
228  {
229  std::cerr << "[SwingFootPlanner::advance] Unable to set the knots for the height "
230  "planner."
231  << std::endl;
232  return false;
233  }
234 
235  if (!this->updateSE3Traj())
236  {
237  std::cerr << "[SwingFootPlanner::advance] Unable to update the SE(3) trajectory."
238  << std::endl;
239  return false;
240  }
241 
242  return true;
243  }
244 
245 
246  // This is the case (3)
247  if (!m_state.isInContact && m_currentTrajectoryTime > m_currentContactPtr->deactivationTime
248  && m_currentTrajectoryTime <= std::next(m_currentContactPtr, 1)->activationTime)
249  {
250  if (!this->updateSE3Traj())
251  {
252  std::cerr << "[SwingFootPlanner::advance] Unable to update the SE(3) trajectory."
253  << std::endl;
254  return false;
255  }
256 
257 
258  return true;
259  }
260 
261  // This is the case (4)
262  if (!m_state.isInContact && m_currentTrajectoryTime > m_currentContactPtr->deactivationTime
263  && m_currentTrajectoryTime > std::next(m_currentContactPtr, 1)->activationTime)
264  {
265  std::advance(m_currentContactPtr, 1);
266 
267  m_state.transform = m_currentContactPtr->pose;
268  m_state.mixedVelocity.setZero();
269  m_state.mixedAcceleration.setZero();
270  m_state.isInContact = true;
271 
272  return true;
273  }
274 
275  return true;
