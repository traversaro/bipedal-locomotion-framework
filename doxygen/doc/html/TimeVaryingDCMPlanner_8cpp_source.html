186  void setupOpti(const std::size_t& horizonSampling)
187  {
188  using Sl = casadi::Slice;
189 
190  // define the size of the optimization variables
191  this->optiVariables.dcm = this->opti.variable(this->dcmVectorSize, horizonSampling + 1);
192  this->optiVariables.omega = this->opti.variable(1, horizonSampling + 1);
193  this->optiVariables.vrp = this->opti.variable(this->vrpVectorSize, horizonSampling);
194  this->optiVariables.omegaDot = this->opti.variable(1, horizonSampling);
195 
196  // set the initial constraints
197  this->optiParameters.dcmInitialPosition = this->opti.parameter(this->dcmVectorSize);
198  this->optiParameters.dcmFinalPosition = this->opti.parameter(this->dcmVectorSize);
199  this->optiParameters.dcmInitialVelocity = this->opti.parameter(this->dcmVectorSize);
200  this->optiParameters.dcmFinalVelocity = this->opti.parameter(this->dcmVectorSize);
201  this->optiParameters.omegaInitialValue = this->opti.parameter();
202  this->optiParameters.omegaFinalValue = this->opti.parameter();
203 
204  // set constraints and cost functions whose equations does not change in the optimization
205  // problem
206  auto& dcm = this->optiVariables.dcm;
207  auto& vrp = this->optiVariables.vrp;
208  auto& omega = this->optiVariables.omega;
209  auto& omegaDot = this->optiVariables.omegaDot;
210 
211  // initial values
212  this->opti.subject_to(omega(Sl(), 0) == this->optiParameters.omegaInitialValue);
213  this->opti.subject_to(dcm(Sl(), 0) == this->optiParameters.dcmInitialPosition);
214 
215 
216  auto initialDcmVelocity = (omega(0) - omegaDot(0) / omega(0)) * (dcm(Sl(), 0) - vrp(Sl(), 0));
217  auto initialOmegaVelocity = omegaDot(0);
218 
219  this->opti.subject_to(casadi::MX::vertcat({initialDcmVelocity, initialOmegaVelocity})
220  == casadi::MX::vertcat({this->optiParameters.dcmInitialVelocity, 0}));
221 
222  // final values
223  this->opti.subject_to(omega(Sl(), -1) == this->optiParameters.omegaFinalValue);
224  this->opti.subject_to(dcm(Sl(), -1) == this->optiParameters.dcmFinalPosition);
225 
226  auto finalDcmVelocity = (omega(-1) - omegaDot(-1) / omega(-1)) * (dcm(Sl(), -1) - vrp(Sl(), -1));
227  auto finalOmegaVelocity = omegaDot(-1);
228 
229  this->opti.subject_to(casadi::MX::vertcat({finalDcmVelocity, finalOmegaVelocity})
230  == casadi::MX::vertcat({this->optiParameters.dcmFinalVelocity, 0}));
231 
232  auto trajectory = this->optiFunctions.dcmDynamics.map(horizonSampling);
233 
234  this->opti.subject_to(
235  casadi::MX::vertcat(
236  {dcm(Sl(), Sl(1, dcm.columns())), omega(Sl(), Sl(1, omega.columns()))})
237  == trajectory({casadi::MX::vertcat({dcm(Sl(), Sl(0, -1)), omega(Sl(), Sl(0, -1))}),
238  vrp,
239  omegaDot})[0]);
240 
241  // omega^2 - omegaDot has to be positive
242  this->opti.subject_to(casadi::MX::pow(omega(Sl(), Sl(0, -1)), 2) - omegaDot
243  > casadi::DM::zeros(omegaDot.rows(), omegaDot.columns()));
244 
245  // omega has to be positive
246  this->opti.subject_to(omega > casadi::DM::zeros(omega.rows(), omega.columns()));
247 
248  this->optiParameters.dcmRefererenceTraj = this->opti.parameter(this->dcmVectorSize, horizonSampling + 1);
249 
250  this->costFunction
251  = this->optiSettings.omegaDotWeight * casadi::MX::sumsqr(this->optiVariables.omegaDot)
252  + this->optiSettings.dcmTrackingWeight
253  * casadi::MX::sumsqr(this->optiVariables.dcm
254  - this->optiParameters.dcmRefererenceTraj)
255  + this->optiSettings.omegaDotRateOfChangeWeight
256  * casadi::MX::sumsqr(casadi::MX::diff(this->optiVariables.omegaDot))
257  + this->optiSettings.vrpRateOfChangeWeight
258  * casadi::MX::sumsqr(casadi::MX::diff(this->optiVariables.vrp.T()))
259  + this->optiSettings.dcmRateOfChangeWeight
260  * casadi::MX::sumsqr(casadi::MX::diff(this->optiVariables.dcm.T()));
261 
262  this->opti.minimize(costFunction);
263 
264  // set the opti object options
265  this->setupOptiOptions();
266  }

317  points.col(columnIndex) = point.head(numberOfCoordinates);

329  ecmpConstraintA = casadi::DM::zeros(A.rows(), A.cols());
330  std::memcpy(ecmpConstraintA.ptr(),
331  A.data(),
332  sizeof(double) * A.rows() * A.cols());
333 
334  const auto& b = this->convexHullHelper.getB();
335  ecmpConstraintB.resize(b.size(), 1);
336  ecmpConstraintB = casadi::DM::zeros(b.size(), 1);

342  bool setupOptimizationProblem(std::shared_ptr<const ContactPhaseList> contactPhaseList,
343  const DCMPlannerState& initialState)
344  {
345  using Sl = casadi::Slice;
346 
347  // set the initial conditions
348  this->opti.set_value(this->optiParameters.dcmInitialPosition,
349  casadi::DM(std::vector<double>(initialState.dcmPosition.data(),
350  initialState.dcmPosition.data()
351  + this->dcmVectorSize)));
352  this->opti.set_value(this->optiParameters.dcmInitialVelocity,
353  casadi::DM(std::vector<double>(initialState.dcmVelocity.data(),
354  initialState.dcmVelocity.data()
355  + this->dcmVectorSize)));
356  this->opti.set_value(this->optiParameters.omegaInitialValue, initialState.omega);
357 
358 
359  double time = contactPhaseList->cbegin()->beginTime;
360 
361  std::vector<Eigen::VectorXd> dcmKnots;
362  std::vector<double> timeKnots;
363 
364  // first point
365  timeKnots.push_back(time);
366  dcmKnots.push_back(initialState.dcmPosition);
367 
368  double averageDCMHeight = initialState.dcmPosition[2];
369 
370  casadi::DM ecmpConstraintA;
371  casadi::DM ecmpConstraintB;
372 
373  const auto& vrp = this->optiVariables.vrp;
374  const auto& omega = this->optiVariables.omega;
375  const auto& omegaDot = this->optiVariables.omegaDot;
376 
377 
378  auto contactPhaseListIt = contactPhaseList->cbegin();
379 
380 
381  bool feetAreInSamePlane = computeConstraintElementsECMP(*contactPhaseListIt,
382  ecmpConstraintA,
383  ecmpConstraintB);
384 
385  for (std::size_t k = 0; k < this->optiVariables.vrp.columns(); k++)
386  {
387  // if the
388  if(k * this->optiSettings.plannerSamplingTime > contactPhaseListIt->endTime)
389  {
390  std::advance(contactPhaseListIt, 1);
391  feetAreInSamePlane = computeConstraintElementsECMP(*contactPhaseListIt,
392  ecmpConstraintA,
393  ecmpConstraintB);
394 
395  if (contactPhaseListIt->activeContacts.size() == 2
396  && contactPhaseListIt != contactPhaseList->begin()
397  && contactPhaseListIt != contactPhaseList->lastPhase())
398  {
399  timeKnots.emplace_back(
400  (contactPhaseListIt->endTime + contactPhaseListIt->beginTime) / 2);
401 
402  auto contactIt = contactPhaseListIt->activeContacts.cbegin();
403  const Eigen::Vector3d p1 = contactIt->second->pose.translation();
404  std::advance(contactIt, 1);
405  const Eigen::Vector3d p2 = contactIt->second->pose.translation();
406 
407  Eigen::Vector3d desiredDCMPosition = (p1 + p2) / 2.0;
408  desiredDCMPosition(2) += averageDCMHeight;
409 
410  dcmKnots.emplace_back(desiredDCMPosition);
411  }
412 
413  // TODO please try to make it more versatile
414  // read it as if the robot is in the last double support phase
415  else if (contactPhaseListIt->activeContacts.size() == 2
416  && contactPhaseListIt == contactPhaseList->lastPhase())
417  {
418  timeKnots.push_back(contactPhaseListIt->endTime);
419  auto contactIt = contactPhaseListIt->activeContacts.cbegin();
420  const Eigen::Vector3d p1 = contactIt->second->pose.translation();
421  std::advance(contactIt, 1);
422  const Eigen::Vector3d p2 = contactIt->second->pose.translation();
423 
424  Eigen::Vector3d desiredDCMPosition = (p1 + p2) / 2.0;
425  desiredDCMPosition(2) += averageDCMHeight;
426 
427  dcmKnots.emplace_back(desiredDCMPosition);
428  }
429  }
430 
431  if (!feetAreInSamePlane)
432  {
433  this->opti.subject_to(
434  casadi::MX::mtimes(ecmpConstraintA,
435  casadi::MX::vertcat(this->optiFunctions.ecmp(
436  {omega(Sl(), k), vrp(Sl(), k), omegaDot(Sl(), k)})))
437  <= ecmpConstraintB);
438  } else
439  {
440  this->opti.subject_to(casadi::MX::mtimes(ecmpConstraintA, vrp(Sl(0, 2), k))
441  <= ecmpConstraintB);
442 
443  this->opti.subject_to(
444  vrp(2, k) - 9.81 / (casadi::MX::pow(omega(Sl(), k), 2) - omegaDot(Sl(), k))
445  == contactPhaseListIt->activeContacts.begin()->second->pose.translation()[2]);
446  }
447  }
448 
449  dcmRef.setInitialConditions(initialState.dcmVelocity, Eigen::Vector3d::Zero());
450  dcmRef.setFinalConditions(Eigen::Vector3d::Zero(), Eigen::Vector3d::Zero());
451  dcmRef.setKnots(dcmKnots, timeKnots);
452 
453  // set last values
454  this->opti.set_value(this->optiParameters.dcmFinalPosition,
455  casadi::DM(std::vector<double>(dcmKnots.back().data(),
456  dcmKnots.back().data()
457  + this->dcmVectorSize)));
458 
459  this->opti.set_value(this->optiParameters.dcmFinalVelocity,
460  casadi::DM::zeros(this->dcmVectorSize));
461 
462  // TODO the final omega is equal to the initial omega
463  this->opti.set_value(this->optiParameters.omegaFinalValue, initialState.omega);
464 
465  casadi::DM initialValueDCM = casadi::DM::zeros(3, this->optiVariables.dcm.columns());
466  casadi::DM initialValueOmega(1, this->optiVariables.dcm.columns());
467 
468  Eigen::Vector3d velocity, acceleration;
469  Eigen::Map<Eigen::MatrixXd> initialValueDCMEigenMap(initialValueDCM.ptr(),
470  initialValueDCM.rows(),
471  initialValueDCM.columns());
472 
473  for (int i = 0; i < this->optiVariables.dcm.columns(); i++)
474  {
475  const double currentTime = i * this->optiSettings.plannerSamplingTime;
476  dcmRef.evaluatePoint(currentTime,
477  initialValueDCMEigenMap.col(i),
478  velocity,
479  acceleration);
480  }
481 
482  initialValueOmega = sqrt(9.81 / initialValueDCM(2,Sl()));
483  this->opti.set_value(this->optiParameters.dcmRefererenceTraj, initialValueDCM);
484 
485  this->opti.set_initial(this->optiVariables.dcm, initialValueDCM);
486  this->opti.set_initial(this->optiVariables.vrp, initialValueDCM(Sl(), Sl(0, -1)));
487  this->opti.set_initial(this->optiVariables.omega, initialValueOmega);
488  this->opti.set_initial(this->optiVariables.omegaDot,
489  casadi::DM::zeros(1, omegaDot.columns()));
490 
491  return true;
492  }
