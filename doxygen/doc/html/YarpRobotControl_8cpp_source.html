408  bool setReferences(Eigen::Ref<const Eigen::VectorXd> jointValues)
409  {
410  constexpr std::string_view errorPrefix = "[YarpRobotControl::Impl::setReferences] ";
411 
412  if(!this->getJointPos())
413  {
414  std::cerr << errorPrefix << "Unable to get the joint position." << std::endl;
415  return false;
416  }
417 
418  const auto worstError = this->getWorstPositionDirectError(jointValues,
419  this->positionFeedback);
420 
421  if (worstError.second > this->positionDirectMaxAdmissibleError)
422  {
423  std::cerr << errorPrefix << "The worst error between the current and the "
424  << "desired position of the joint named '" << this->axesName[worstError.first]
425  << "' is greater than " << this->positionDirectMaxAdmissibleError
426  << " rad. Error = " << worstError.second << " rad." << std::endl;
427  return false;
428  }
429 
430  for (const auto& [mode, indeces] : this->desiredJointValuesAndMode.index)
431  {
432  // if indeces vector is empty no joint is controlled with this specific control mode
433  if (indeces.empty())
434  continue;
435 
436  if (mode == IRobotControl::ControlMode::Idle)
437  continue;
438 
439  else if (mode == IRobotControl::ControlMode::Unknown)
440  {
441  std::string error = " The following joints does not have a specified control "
442  "mode: ";
443 
444  for (const auto& index : indeces)
445  error += "'" + this->axesName[index] + "' ";
446 
447  std::cerr << errorPrefix << error << ". Please set a feasible control mode."
448  << std::endl;
449 
450  return false;
451 
452  } else if (mode == IRobotControl::ControlMode::Position)
453  {
454  std::vector<double> refSpeeds(indeces.size());
455  for (int i = 0; i < indeces.size(); i++)
456  {
457  const auto jointError = std::abs(jointValues[indeces[i]]
458  - this->positionFeedback[indeces[i]]);
459 
460  constexpr double scaling = 180 / M_PI;
461  constexpr double maxVelocityInDegPerSeconds = 3.0;
462  refSpeeds[i] = std::max(maxVelocityInDegPerSeconds,
463  scaling * (jointError / this->positioningDuration));
464 
465  this->positionInterface->setRefSpeeds(indeces.size(),
466  indeces.data(),
467  refSpeeds.data());
468  }
469 
470  this->startPositionControlInstant = yarp::os::Time::now();
471  }
472 
473  // Yarp wants the quantities in degrees
474  double scaling = 180 / M_PI;
475  if (mode == ControlMode::Torque)
476  scaling = 1;
477 
478  for (int i = 0; i < indeces.size(); i++)
479  this->desiredJointValuesAndMode.value[mode][i] = scaling * jointValues[indeces[i]];
480 
481  if (!this->control(mode)(indeces.size(),
482  indeces.data(),
483  this->desiredJointValuesAndMode.value[mode].data()))
484 
485  {
486  std::cerr << errorPrefix << "Unable to set the desired joint values." << std::endl;
487  return false;
488  }
489  }
490  return true;
491  }
